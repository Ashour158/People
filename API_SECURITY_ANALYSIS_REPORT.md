```python
# CSRF protection middleware
class CSRFProtectionMiddleware(BaseHTTPMiddleware):
    def generate_csrf_token(self, session_id: str) -> str:
        message = f"{session_id}:{secrets.token_urlsafe(32)}"
        token = hmac.new(
            self.secret_key,
            message.encode(),
            hashlib.sha256
        ).hexdigest()
        return token
```
